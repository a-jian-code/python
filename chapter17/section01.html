Python基础

连接数据库

使用Python代码，再结合Python第三方扩展库，可以操作多种数据库。

操作sqlite数据库

sqlite3数据库是一种小型的数据库类型，通常用于移动设备(如手机等)。不需要安装额外的扩展库，Python自带的sqlite3模块可以直接来读取sqlite3数据库。

import sqlite3

# 使用Python自带的 sqlite3 包，和 sqlite 数据库建立一个链接
conn = sqlite3.Connection('./data/data.sqlite')

# 调用 execute 方法，执行 sql 语句，操作服务器
cursor = conn.execute('select * from student')
data = cursor.fetchmany(100)
for d in data:
    print(d)

使用pymysql和mysqlclient操作数据库

Python自带模块不能直接操作mysql数据，需要安装额外的第三发库来实现mysql数据库的读写操作。常见的mysql数据库插件有pymysql和mysqlclient，以及sqlalchemy. 其中，pymysql和mysqlclient功能类似。

import pymsql
import MySQLdb  # mysqlclient 安装完成以后的包名是 MySQLdb

db = pymysql.Connection(host='localhost', port=3306, database='test', charset='utf8', user='root', password='abcd1234')
db.begin()

cursor = db.cursor()
drop_sql = 'drop table if exists student'
cursor.execute(drop_sql)
create_sql = "create table student (" \
             "id int primary key auto_increment," \
             "name varchar(32) not null unique," \
             "age tinyint unsigned) charset=utf8"
cursor.execute(create_sql)

insert_sql = "insert into student(name,age) values('zhangsan',18),('lisi',20),('jack',19)"
cursor.execute(insert_sql)

query_sql = 'select * from student where name="zhangsan"'
cursor.execute(query_sql)
print(cursor.fetchall())

db.commit()
db.close()

sqlalchemy的使用

sqlalchemy是Python里最常用的一款ORM(Object-Relational Mapping,对象关系映射)框架。

pymysql操作数据有一下几个问题需要解决:

  • 所有的操作都是使用 sql 语句执行，不能直接使用Python面向对象的操作实现。
  • 查询获取到的结果都是类似于 ((1, 'zhangsan', 18), (2, 'lisi', 20), (3, 'jack', 19))这种元组嵌套的结果，操作不便。

ORM的作用就是将数据库里的字段和Python里的对象进行关联，直接操作Python对象就可以修改数据库，对数据库进行查询，得到的结果也是一个Python对象，操作起来更加的方便。

import sqlalchemy
from sqlalchemy import orm, Column, String, Integer
from sqlalchemy.ext.declarative import declarative_base

Base = declarative_base()


class Student(Base):
    __tablename__ = 'student'
    id = Column(Integer, primary_key=True)
    name = Column(String(32), nullable=False, unique=True)
    age = Column(Integer)

    def __repr__(self):
        return f'{self.id}-{self.name}-{self.age}'


engine = sqlalchemy.create_engine('mysql+pymysql://root:abcd1234@localhost:3306/test', encoding='utf-8')
session = orm.sessionmaker(bind=engine)()
print(type(session))
# 查询数据
print(session.query(Student).all())
print(session.query(Student).filter_by(name="lisi").first())
print(session.query(Student).order_by('age').all())
print(session.query(Student).offset(1).all())
print(session.query(Student).limit(2).all())

# 更新数据
session.query(Student).filter_by(name='lisi').update({'age': Student.age + 2})

# 添加数据
s1 = Student(id=4, name='tony', age=19)
s2 = Student(id=5, name='jerry', age=20)
s3 = Student(id=6, name='henry', age=18)

session.add(s1)
session.add_all([s2, s3])

# 删除数据
student = session.query(Student).filter_by(name='lisi').first()
session.delete(student)
print(session.query(Student).all())
session.commit()
session.close()
